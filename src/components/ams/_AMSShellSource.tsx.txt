// ───── Mega Menu Dropdown (viewport-safe) ─────
function MegaMenu({ group, activeNav, onPick, soundOn }: {
  group: NavGroup; activeNav: string; onPick: (l: string) => void; soundOn: boolean;
}) {
  const [open, setOpen] = useState(false);
  const [pos, setPos] = useState<{ left: number; top: number; maxH: number; w: number }>({ left: 0, top: 0, maxH: 480, w: 420 });
  const ref = useRef<HTMLDivElement>(null);
  const btnRef = useRef<HTMLButtonElement>(null);
  const Icon = group.icon;
  const contains = group.items.some((i) => i.label === activeNav);

  useEffect(() => {
    function onDoc(e: MouseEvent) { if (!ref.current?.contains(e.target as Node)) setOpen(false); }
    document.addEventListener("click", onDoc);
    return () => document.removeEventListener("click", onDoc);
  }, []);

  // measure + clamp to viewport on open / resize
  useEffect(() => {
    if (!open) return;
    function place() {
      const btn = btnRef.current; if (!btn) return;
      const r = btn.getBoundingClientRect();
      const PAD = 12;
      const vw = window.innerWidth, vh = window.innerHeight;
      const w = Math.min(420, vw - PAD * 2);
      let left = r.left;
      if (left + w + PAD > vw) left = Math.max(PAD, vw - w - PAD);
      const top = r.bottom + 8;
      const maxH = Math.max(220, vh - top - PAD);
      setPos({ left, top, w, maxH });
    }
    place();
    window.addEventListener("resize", place);
    window.addEventListener("scroll", place, true);
    return () => { window.removeEventListener("resize", place); window.removeEventListener("scroll", place, true); };
  }, [open]);

  return (
    <div ref={ref} className="relative">
      <button
        ref={btnRef}
        onClick={() => { setOpen((o) => !o); if (soundOn) playCoin(); }}
        className={`group flex items-center gap-2 rounded-md px-3 py-2 text-xs font-semibold uppercase tracking-[0.16em] transition-all ${
          contains ? "bg-[oklch(0.78_0.14_82/0.14)] text-[#f5d77a] shadow-[inset_0_0_0_1px_oklch(0.78_0.14_82/0.45)]"
          : "text-muted-foreground hover:bg-[oklch(0.78_0.14_82/0.08)] hover:text-foreground"
        }`}
      >
        <Icon className="h-3.5 w-3.5" />
        <span>{group.label}</span>
        <ChevronDown className={`h-3 w-3 transition-transform ${open ? "rotate-180" : ""}`} />
      </button>
      {open && (
        <div
          style={{ position: "fixed", left: pos.left, top: pos.top, width: pos.w, maxHeight: pos.maxH }}
          className="z-50 origin-top-left animate-scale-in overflow-hidden rounded-xl border border-gold bg-[oklch(0.13_0.025_250)]/95 shadow-[0_24px_60px_-20px_rgba(0,0,0,0.8),inset_0_1px_0_oklch(0.78_0.14_82/0.2)] backdrop-blur-xl"
        >
          <div className="flex items-center justify-between border-b border-gold p-4 pb-3">
            <div>
              <div className="font-display text-base font-semibold text-gold-gradient">{group.label}</div>
              <div className="text-[9px] uppercase tracking-[0.25em] text-[#d4a14a]">{group.tagline}</div>
            </div>
            <Icon className="h-5 w-5 shrink-0 text-[#f5d77a]" />
          </div>
          <div className="grid gap-1 overflow-y-auto p-2" style={{ maxHeight: pos.maxH - 70 }}>
            {group.items.map((item) => {
              const ItIcon = item.icon;
              const active = item.label === activeNav;
              return (
                <button
                  key={item.label}
                  onClick={() => { onPick(item.label); setOpen(false); if (soundOn) playCoin(); }}
                  className={`flex items-start gap-3 rounded-lg px-3 py-2.5 text-left transition-all ${
                    active ? "bg-gold-gradient text-[oklch(0.13_0.025_250)]"
                    : "text-foreground/90 hover:bg-[oklch(0.78_0.14_82/0.1)]"
                  }`}
                >
                  <ItIcon className={`mt-0.5 h-4 w-4 shrink-0 ${active ? "text-[oklch(0.13_0.025_250)]" : "text-[#f5d77a]"}`} />
                  <div className="min-w-0">
                    <div className="text-sm font-semibold">{item.label}</div>
                    <div className={`text-[11px] ${active ? "text-[oklch(0.13_0.025_250)]/80" : "text-muted-foreground"}`}>{item.desc}</div>
                  </div>
                </button>
              );
            })}
          </div>
        </div>
      )}
    </div>
  );
}


// ───── Command Palette (Linear / Raycast / Vercel) ─────
function CommandPalette({ open, setOpen, onPick }: {
  open: boolean; setOpen: (b: boolean) => void; onPick: (l: string) => void;
}) {
  const [q, setQ] = useState("");
  const [idx, setIdx] = useState(0);
  const inputRef = useRef<HTMLInputElement>(null);

  const results = useMemo(() => {
    const needle = q.trim().toLowerCase();
    if (!needle) return ALL_ITEMS.slice(0, 8);
    return ALL_ITEMS.filter((i) => i.label.toLowerCase().includes(needle) || i.desc.toLowerCase().includes(needle)).slice(0, 12);
  }, [q]);

  useEffect(() => { setIdx(0); }, [q, open]);
  useEffect(() => { if (open) setTimeout(() => inputRef.current?.focus(), 30); }, [open]);

  if (!open) return null;
  return (
    <div className="fixed inset-0 z-[100] flex items-start justify-center bg-[oklch(0.08_0.02_250/0.7)] p-4 pt-[12vh] backdrop-blur-sm" onClick={() => setOpen(false)}>
      <div className="w-full max-w-xl overflow-hidden rounded-2xl border border-gold bg-[oklch(0.13_0.025_250)] shadow-[0_40px_120px_-20px_rgba(0,0,0,0.9)]" onClick={(e) => e.stopPropagation()}>
        <div className="flex items-center gap-3 border-b border-gold px-4 py-3">
          <Search className="h-4 w-4 text-[#f5d77a]" />
          <input
            ref={inputRef} value={q} onChange={(e) => setQ(e.target.value)}
            onKeyDown={(e) => {
              if (e.key === "Escape") setOpen(false);
              if (e.key === "ArrowDown") { e.preventDefault(); setIdx((i) => Math.min(results.length - 1, i + 1)); }
              if (e.key === "ArrowUp") { e.preventDefault(); setIdx((i) => Math.max(0, i - 1)); }
              if (e.key === "Enter" && results[idx]) { onPick(results[idx].label); setOpen(false); }
            }}
            placeholder="Search AMS — jump to any screen, action, or member…"
            className="flex-1 bg-transparent text-sm outline-none placeholder:text-muted-foreground/70"
          />
          <kbd className="rounded border border-gold px-2 py-0.5 text-[10px] text-muted-foreground">ESC</kbd>
        </div>
        <div className="max-h-[50vh] overflow-y-auto p-2">
          {results.length === 0 && <div className="px-3 py-8 text-center text-sm text-muted-foreground">No results for "{q}"</div>}
          {results.map((r, i) => {
            const Ic = r.icon;
            const active = i === idx;
            return (
              <button key={r.label} onMouseEnter={() => setIdx(i)} onClick={() => { onPick(r.label); setOpen(false); }}
                className={`flex w-full items-center gap-3 rounded-lg px-3 py-2.5 text-left ${active ? "bg-[oklch(0.78_0.14_82/0.15)] text-[#f5d77a]" : "text-foreground/90"}`}>
                <Ic className="h-4 w-4 shrink-0 text-[#f5d77a]" />
                <div className="min-w-0 flex-1"><div className="text-sm font-medium">{r.label}</div><div className="truncate text-[11px] text-muted-foreground">{r.desc}</div></div>
                {active && <span className="text-[10px] text-muted-foreground">↵</span>}
              </button>
            );
          })}
        </div>
        <div className="flex items-center justify-between border-t border-gold px-4 py-2 text-[10px] uppercase tracking-[0.18em] text-muted-foreground">
          <div className="flex gap-3"><span>↑↓ navigate</span><span>↵ open</span></div>
          <div className="flex items-center gap-1"><Command className="h-3 w-3" />K to toggle</div>
        </div>
      </div>
    </div>
  );
}

// ───── Particles ─────
function Particles({ count = 18 }: { count?: number }) {
  const seeds = useMemo(() => Array.from({ length: count }, (_, i) => ({
    left: (i * 53) % 100, delay: (i * 0.37) % 6, duration: 4 + ((i * 1.3) % 4), size: 2 + (i % 4),
  })), [count]);
  return (
    <div className="pointer-events-none absolute inset-0 overflow-hidden">
      {seeds.map((s, i) => (
        <span key={i} className="gold-particle" style={{ left: `${s.left}%`, bottom: 0, width: s.size, height: s.size, animationDelay: `${s.delay}s`, animationDuration: `${s.duration}s` }} />
      ))}
    </div>
  );
}

function ScoreRing({ value, label }: { value: number; label: string }) {
  const r = 32, c = 2 * Math.PI * r;
  const off = c - (value / 100) * c;
  return (
    <div className="flex flex-col items-center gap-2">
      <div className="relative h-20 w-20">
        <svg viewBox="0 0 80 80" className="h-full w-full -rotate-90">
          <circle cx="40" cy="40" r={r} stroke="oklch(0.3 0.04 250)" strokeWidth="6" fill="none" />
          <circle cx="40" cy="40" r={r} stroke="#f5d77a" strokeWidth="6" fill="none" strokeDasharray={c} strokeDashoffset={off} strokeLinecap="round" style={{ filter: "drop-shadow(0 0 6px #f5d77a)" }} />
        </svg>
        <div className="absolute inset-0 flex items-center justify-center"><span className="font-display text-xl font-semibold text-gold-gradient">{value}</span></div>
      </div>
      <span className="text-[11px] uppercase tracking-[0.18em] text-muted-foreground">{label}</span>
    </div>
  );
}

function Shell() {
  return (
    <CelebrationProvider>
      <ShellInner />
    </CelebrationProvider>
  );
}

function ShellInner() {
  const [activeNav, setActiveNav] = useState("Dashboard");
  const [cmdOpen, setCmdOpen] = useState(false);
  const { celebrate, surprise, soundOn, setSoundOn } = useCelebration();

  useEffect(() => {
    const t = setTimeout(() => { if (soundOn) playUnveil(); }, 600);
    return () => clearTimeout(t);
    // eslint-disable-next-line react-hooks/exhaustive-deps
  }, []);

  useEffect(() => {
    function onKey(e: KeyboardEvent) {
      if ((e.metaKey || e.ctrlKey) && e.key.toLowerCase() === "k") { e.preventDefault(); setCmdOpen((o) => !o); }
    }
    window.addEventListener("keydown", onKey);
    return () => window.removeEventListener("keydown", onKey);
  }, []);

  // Scroll to top on screen change
  useEffect(() => { window.scrollTo({ top: 0, behavior: "smooth" }); }, [activeNav]);

  const ScreenComp = SCREENS[activeNav];

  return (
    <div className="min-h-screen text-foreground">
      <CommandPalette open={cmdOpen} setOpen={setCmdOpen} onPick={setActiveNav} />

      <header className="sticky top-0 z-30 border-b border-gold bg-[oklch(0.13_0.025_250/0.92)] backdrop-blur-md">
        <div className="flex items-center gap-4 px-4 py-3 md:px-8">
          {/* Brand — click for surprise reward */}
          <button
            onClick={surprise}
            onDoubleClick={() => setActiveNav("Dashboard")}
            title="Click for a surprise reward · Double-click to go home"
            className="flex shrink-0 items-center gap-3 transition-transform hover:scale-[1.03]"
          >
            <div className="grid h-10 w-10 place-items-center rounded-lg bg-gold-gradient text-[oklch(0.13_0.025_250)] shadow-[0_8px_24px_-8px_#d4a14a]">
              <Trophy className="h-5 w-5" strokeWidth={2.4} />
            </div>
            <div className="text-left">
              <div className="font-display text-xl font-semibold text-gold-gradient leading-none">AMS</div>
              <div className="mt-1 text-[9px] uppercase tracking-[0.25em] text-muted-foreground">Software Vala</div>
            </div>
          </button>

          {/* Dashboard pill */}
          <button onClick={() => setActiveNav("Dashboard")}
            className={`ml-2 hidden items-center gap-2 rounded-md px-3 py-2 text-xs font-semibold uppercase tracking-[0.16em] lg:flex ${
              activeNav === "Dashboard" ? "bg-[oklch(0.78_0.14_82/0.14)] text-[#f5d77a] shadow-[inset_0_0_0_1px_oklch(0.78_0.14_82/0.45)]" : "text-muted-foreground hover:bg-[oklch(0.78_0.14_82/0.08)] hover:text-foreground"
            }`}>
            <LayoutDashboard className="h-3.5 w-3.5" /> Dashboard
          </button>

          {/* Mega menus */}
          <nav className="hidden items-center gap-1 lg:flex">
            {NAV.map((g) => <MegaMenu key={g.label} group={g} activeNav={activeNav} onPick={setActiveNav} soundOn={soundOn} />)}
          </nav>

          {/* Mobile: single grouped */}
          <div className="ml-2 lg:hidden">
            <MegaMenu group={{ label: "Modules", icon: Layers, tagline: "All AMS screens", items: ALL_ITEMS }} activeNav={activeNav} onPick={setActiveNav} soundOn={soundOn} />
          </div>

          <div className="ml-auto flex items-center gap-2">
            {/* Celebrate demo */}
            <button
              onClick={() => celebrate({ kind: "rankUp", title: "Promoted to Champion", subtitle: "You crossed 200,000 XP — Rank #4 globally.", rarity: "Legendary", xp: 5000 })}
              className="hidden items-center gap-2 rounded-full bg-gold-gradient px-3 py-1.5 text-xs font-semibold text-[oklch(0.13_0.025_250)] shadow-[0_8px_20px_-8px_#d4a14a] hover:scale-105 transition-transform md:inline-flex"
              title="Trigger a premium celebration"
            >
              <Sparkles className="h-3.5 w-3.5" /> Celebrate
            </button>

            <button onClick={() => setCmdOpen(true)} className="hidden items-center gap-2 rounded-full border border-gold bg-[oklch(0.18_0.03_250)] px-3 py-1.5 text-sm text-muted-foreground hover:bg-[oklch(0.78_0.14_82/0.08)] xl:flex">
              <Search className="h-4 w-4" /><span className="w-44 text-left">Search AMS…</span>
              <kbd className="rounded border border-gold px-1.5 py-0.5 text-[10px]">⌘K</kbd>
            </button>
            <button onClick={() => setCmdOpen(true)} className="grid h-9 w-9 place-items-center rounded-full border border-gold bg-[oklch(0.18_0.03_250)] text-[#f5d77a] xl:hidden"><Search className="h-4 w-4" /></button>
            <button onClick={() => setSoundOn(!soundOn)} title={soundOn ? "Mute" : "Unmute"} className="grid h-9 w-9 place-items-center rounded-full border border-gold bg-[oklch(0.18_0.03_250)] text-[#f5d77a] hover:bg-[oklch(0.78_0.14_82/0.15)]">
              {soundOn ? <Volume2 className="h-4 w-4" /> : <VolumeX className="h-4 w-4 opacity-60" />}
            </button>
            <button className="grid h-9 w-9 place-items-center rounded-full border border-gold bg-[oklch(0.18_0.03_250)] text-[#f5d77a]"><Bell className="h-4 w-4" /></button>
            <button onClick={() => setActiveNav("Profile")} className="flex items-center gap-2 rounded-full border border-gold bg-[oklch(0.18_0.03_250)] py-1 pl-1 pr-3 hover:bg-[oklch(0.78_0.14_82/0.08)]">
              <div className="grid h-7 w-7 place-items-center rounded-full bg-gold-gradient text-[oklch(0.13_0.025_250)] text-[10px] font-bold">{ME.initials}</div>
              <div className="hidden text-left sm:block">
                <div className="text-xs font-semibold leading-none">{ME.name}</div>
                <div className="mt-0.5 text-[9px] uppercase tracking-[0.2em] text-[#d4a14a]">{ME.tier} · Rank {ME.rank}</div>
              </div>
            </button>
          </div>
        </div>
      </header>

      <main className="px-6 py-8 md:px-10">
        {activeNav === "Dashboard" || !ScreenComp ? (
          <DashboardHome onJump={setActiveNav} soundOn={soundOn} />
        ) : (
          (() => {
            const group = NAV.find((g) => g.items.some((i) => i.label === activeNav));
            const siblings = group?.items.map((i) => i.label) ?? [];
            const idx = siblings.indexOf(activeNav);
            const next = idx >= 0 && idx < siblings.length - 1 ? siblings[idx + 1] : undefined;
            return (
              <Workspace
                module={group?.label ?? "AMS"}
                screen={activeNav}
                related={siblings}
                next={next}
                onJump={setActiveNav}
                onHome={() => setActiveNav("Dashboard")}
              >
                <ScreenComp />
              </Workspace>
            );
          })()
        )}
        <footer className="mt-12 border-t border-gold pt-6 text-center text-[10px] uppercase tracking-[0.3em] text-muted-foreground">
          Powered By <span className="text-[#f5d77a]">Software Vala AMS</span> · Achievement Management System
        </footer>
      </main>
    </div>
  );
}
